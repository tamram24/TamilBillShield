# ── Knowledge base tests ──────────────────────────────────────────────────────

class TestKnowledgeBase:
    def test_cghs_lookup_exact_match(self):
        from ingestion.knowledge_base import lookup_cghs_rate
        rate, key = lookup_cghs_rate("CT Scan Chest")
        assert rate == 2500.0
        assert "CT Scan" in key

    def test_cghs_lookup_fuzzy_match(self):
        from ingestion.knowledge_base import lookup_cghs_rate
        # Slightly different name — should still match
        rate, key = lookup_cghs_rate("CT Chest Scan")
        assert rate is not None

    def test_cghs_lookup_no_match(self):
        from ingestion.knowledge_base import lookup_cghs_rate
        rate, key = lookup_cghs_rate("xyzzy_completely_unknown_procedure_12345")
        assert rate is None
        assert key is None

    def test_nppa_lookup_found(self):
        from ingestion.knowledge_base import lookup_nppa_cap
        cap = lookup_nppa_cap("Metformin 500mg Tab")
        assert cap == 2.5

    def test_nppa_lookup_not_found(self):
        from ingestion.knowledge_base import lookup_nppa_cap
        cap = lookup_nppa_cap("Unicorn Dust 100mg")
        assert cap is None

    def test_irdai_rules_loaded(self):
        from ingestion.knowledge_base import get_irdai_rules
        rules = get_irdai_rules()
        assert len(rules) > 0
        assert all("rule" in r for r in rules)
        assert all("circular_ref" in r for r in rules)


# ── Model validation tests ────────────────────────────────────────────────────

class TestModels:
    def test_bill_line_item_defaults(self):
        item = BillLineItem(description="Test item", billed_amount=1000.0)
        assert item.confidence == ConfidenceLevel.UNVERIFIABLE
        assert item.cghs_rate is None
        assert item.overcharge_amount is None

    def test_parsed_bill_creation(self, sample_bill):
        assert sample_bill.patient_demand == 193000.0
        assert len(sample_bill.line_items) == 5

    def test_confidence_level_values(self):
        assert ConfidenceLevel.VERIFIED_OVERCHARGE == "verified_overcharge"
        assert ConfidenceLevel.SUSPECTED_OVERCHARGE == "suspected_overcharge"
        assert ConfidenceLevel.UNVERIFIABLE == "unverifiable"
        assert ConfidenceLevel.APPEARS_FAIR == "appears_fair"

    def test_full_analysis_contestable_amount(
        self, sample_bill, sample_discharge, sample_policy
    ):
        from shared.models import (
            AuditResult,
            ContradictionResult,
            FullAnalysis,
            PolicyCheckResult,
        )

        audit = AuditResult(
            verified_overcharge_total=50000.0,
            suspected_overcharge_total=30000.0,
            unverifiable_total=20000.0,
            line_items=[],
            summary="Test summary",
        )
        policy_check = PolicyCheckResult(
            wrongful_rejections=["Non-payable items wrongfully rejected"],
            valid_rejections=[],
            irdai_violations=["IRDAI/HLT/REG/CIR/170/06/2025 Clause 4.1"],
            recoverable_amount=40000.0,
            summary="Test policy summary",
        )
        contradictions = ContradictionResult(
            contradictions_found=True,
            specialist_discrepancy="bill: 8 consultations, discharge: 2 doctors",
            summary="Significant discrepancy found",
        )
        analysis = FullAnalysis(
            parsed_bill=sample_bill,
            parsed_policy=sample_policy,
            audit=audit,
            policy_check=policy_check,
            contradictions=contradictions,
            total_contestable_amount=90000.0,
        )
        assert analysis.total_contestable_amount == 90000.0
        assert analysis.contradictions.contradictions_found is True


# ── OCR tests (mocked) ────────────────────────────────────────────────────────

class TestOCR:
    def test_unsupported_file_type_raises(self):
        from ingestion.ocr import extract_text_from_file
        with pytest.raises(ValueError, match="Unsupported file type"):
            extract_text_from_file(b"data", "document.docx")

    @patch("ingestion.ocr.client")
    def test_image_extraction_calls_claude(self, mock_client):
        mock_response = MagicMock()
        mock_response.content = [MagicMock(text="Extracted text from image")]
        mock_client.messages.create.return_value = mock_response

        from ingestion.ocr import _extract_from_image
        result = _extract_from_image(b"fake_image_bytes", ".png")
        assert result == "Extracted text from image"
        mock_client.messages.create.assert_called_once()

# ── Contradiction detector tests (mocked) ────────────────────────────────────

class TestContradictionDetector:
    @patch("reasoning.contradiction_detector.client")
    def test_detects_specialist_mismatch(
        self, mock_client, sample_bill, sample_discharge
    ):
        mock_result = {
            "specialist_discrepancy": "bill: 8 consultations, discharge: 2 doctors",
            "medicine_discrepancy": None,
            "procedure_discrepancy": None,
            "contradictions_found": True,
            "summary": "Significant specialist discrepancy found.",
        }
        mock_response = MagicMock()
        mock_response.content = [MagicMock(text=json.dumps(mock_result))]
        mock_client.messages.create.return_value = mock_response

        from reasoning.contradiction_detector import detect_contradictions
        result = detect_contradictions(sample_bill, sample_discharge)
        assert result.contradictions_found is True
        assert result.specialist_discrepancy is not None

    @patch("reasoning.contradiction_detector.client")
    def test_no_contradictions(self, mock_client, sample_bill, sample_discharge):
        mock_result = {
            "specialist_discrepancy": None,
            "medicine_discrepancy": None,
            "procedure_discrepancy": None,
            "contradictions_found": False,
            "summary": "No significant contradictions found.",
        }
        mock_response = MagicMock()
        mock_response.content = [MagicMock(text=json.dumps(mock_result))]
        mock_client.messages.create.return_value = mock_response

        from reasoning.contradiction_detector import detect_contradictions
        result = detect_contradictions(sample_bill, sample_discharge)
        assert result.contradictions_found is False
